DROP TABLE Magaz_Proekt.Order;
DROP TABLE Magaz_Proekt.Backet;
--товары
CREATE TABLE Magaz_Proekt.product(
  RootId CHARACTER VARYING(30) UNIQUE NOT NULL,
  Name CHARACTER VARYING(90) NOT NULL,
  DimId CHARACTER VARYING(10)  NULL,
  Price numeric NOT NULL,
  GroupId CHARACTER VARYING(30) NOT NULL,
  WEIGHT numeric NULL,
  postname CHARACTER VARYING(20) NULL,
  Remark CHARACTER VARYING(240) NULL
);
--статус юзера
CREATE TABLE Magaz_Proekt.Status (
  id   INTEGER UNIQUE NOT NULL,
  Status CHARACTER VARYING (30)UNIQUE NOT NULL
);
--покупатель
CREATE TABLE Magaz_Proekt.user (
id   SERIAL PRIMARY KEY,
name CHARACTER VARYING(128) UNIQUE NOT NULL,
email CHARACTER VARYING (100) UNIQUE NOT NULL,
Status INTEGER REFERENCES Magaz_Proekt.Status(id) NOT NULL
);
--корзина
CREATE TABLE Magaz_Proekt.Backet (
  id_order SERIAL PRIMARY KEY,
  id_user INTEGER  REFERENCES Magaz_Proekt.user(id) NOT NULL
);
CREATE TABLE Magaz_Proekt.Order (
  id_order INTEGER  REFERENCES Magaz_Proekt.Backet(id_order) NOT NULL,
  id_product CHARACTER VARYING(30) REFERENCES Magaz_Proekt.product(RootId) NOT NULL
);
